@startuml
class ModelInput
{
  + Label: float
  + Feature: string
}
class Prediction
{
  + PredictedLabel: float
  + Score: float[]
}
enum DataViewType
{
  + File
  + List
}
interface IWebsiteAIAssistantLogger
{
  LogInformation(string message, params object[] args): void
  LogWarning(string message, params object[] args): void
  LogError(string message, params object[] args): void
}

class TextFeaturizingEstimatorOptions
{
  + CaseMode: TextFeaturizingEstimatorCaseMode
  + KeepDiacritics: bool
  + KeepPunctuations: bool
  + KeepNumbers: bool
  + CharFeatureExtractor: WordBagEstimatorOptions
  + WordFeatureExtractor: WordBagEstimatorOptions
}

class WordBagEstimatorOptions
{
  + NgramLength: int
  + UseAllLengths: bool
  + MaximumNgramsCount: int[]
  + SkipLength: int
  + Weighting: WordBagWeightingCriteria
}

enum WordBagWeightingCriteria
{
  Tf
  Idf
  TfIdf
}

enum TextFeaturizingEstimatorCaseMode
{
  Lower
  Upper
  None
}

class SdcaMaximumEntropyOptions
{
  + BiasLearningRate: float
  + ConvergenceCheckFrequency: int?
  + MaximumNumberOfIterations: int?
  + ConvergenceTolerance: float
  + L1Regularization: float?
  + L2Regularization: float?
  + Shuffle: bool
  + NumberOfThreads: int?
}

class PredictionEngine
{
  + DataViewType: DataViewType
  + DataViewFilePath: string
  + DataViewList: IEnumerable<ModelInput>
  + NegativeConfidenceThreshold: float
  + NegativeLabel: float
  + StopWords: string[]
  + TextFeaturizingEstimatorOptions: TextFeaturizingEstimatorOptions
  + SdcaMaximumEntropyOptions: SdcaMaximumEntropyOptions
  + ExtendedFeatureColumnNames: string[]
  + AIModelLoadFilePath: string
  + IsPredictionEngineInitialized: bool
  + Logger: IWebsiteAIAssistantLogger
  + CreateModelAsync(string modelPath): Task
  + CreateModelAsync<TModelInput>(string modelPath): Task
  + LoadModelAsync(string modelPath): Task<bool>
  + LoadModelAsync(): Task<bool>
  + UnloadModelAsync(): Task<bool>
  + ResetAsync(): Task
  + PredictAsync(ModelInput input): Task<Prediction>
}
